CREATE TABLE patrons (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    birthday DATE NOT NULL,
    sex ENUM('male', 'female', 'other', 'not_specified') NOT NULL,
    house_number VARCHAR(50) NOT NULL,
    country VARCHAR(100) NOT NULL,
    mobile_phone VARCHAR(20) NOT NULL UNIQUE,
    secondary_phone VARCHAR(20),
    other_phone VARCHAR(20),
    email VARCHAR(255) NOT NULL UNIQUE,
    secondary_email VARCHAR(255) UNIQUE,
    primary_contact_method ENUM('mobile_phone', 'secondary_phone', 'other_phone', 'email', 'secondary_email') NOT NULL,
    card_number VARCHAR(50) NOT NULL UNIQUE,
    college VARCHAR(255) NOT NULL,
    course VARCHAR(255) NOT NULL,
    registration_date DATE NOT NULL,
    expiry_date DATE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
